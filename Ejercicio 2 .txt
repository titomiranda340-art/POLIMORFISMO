class Animal:
    def hacerSonido(self):
        raise NotImplementedError

class Perro(Animal):
    def hacerSonido(self):
        return "Guau"

class Gato(Animal):
    def hacerSonido(self):
        return "Miau"

def main():
    animales = [
        Perro(),
        Gato()
    ]

    for animal in animales:
        print(f"{animal.__class__.__name__}: {animal.hacerSonido()}")

if __name__ == "__main__":
    main()
